procedure readmatrix(reference matrix: m):
	integer : i;
	integer : j;
begin
	i := 1;
	while i <= 2 do
		j := 1;
		while j <= 2 do
			read m(i,j);
			j := j + 1
		endwhile;
		i := i+1
	endwhile
end;

	integer : i, j;
begin

		j: =1;
